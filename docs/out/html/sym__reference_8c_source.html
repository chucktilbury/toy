32static inline void validate_assignment_reference(ast_assignment_t* node) {
33
34 ENTER;
35
36 int rtype = node->assignment_right->type;
37
38 token_t* tok = node->IDENTIFIER;
39 TRACE("identifier = %s %d %d", tok->raw, tok->line_no, tok->col_no);
40 TRACE("ctx = %p", (void*)ctx);
41 symbol_t* sym = find_symbol(ctx, tok->raw);
42 if(sym != NULL) {
43 node->type = check_assignment_type(tok, sym->sym_type, rtype);
44 sym->ref_count++;
45 TRACE("symbol \"%s\" of type %s:%s is found", tok->raw,
46 (sym->sym_class) ? "DATA" : "FUNC", token_to_str(sym->sym_type));
47 }
48 else
49 syntax_error(tok, "in assignment to \"%s\", symbol not defined", tok->raw);
50
51 RETURN();
52}

54static inline void validate_func_reference(ast_func_reference_t* node) {
55
56 ENTER;

58 token_t* tok = node->IDENTIFIER;
59 TRACE("ctx = %p", (void*)ctx);
60 symbol_t* sym = find_symbol(ctx, tok->raw);
61 if(sym != NULL) {
62 if(sym->sym_class != SYM_FUNC)
63 syntax_error(tok, "symbol \"%s\" is not a function", tok->raw);
64 else {
65 node->type = sym->sym_type;
66 sym->ref_count++;
67
68 TRACE("symbol \"%s\" of type %s:%s is found", tok->raw,
69 (sym->sym_class) ? "DATA" : "FUNC", token_to_str(sym->sym_type));
70 }
71 }
72 else
73 syntax_error(tok, "function definition for \"%s\" not found", tok->raw);
74

78static inline void set_expression_type(ast_expression_t* node) {
79
80 ENTER;
81
82 if(node->expr_primary != NULL) {
83 node->type = node->expr_primary->type;
84 TRACE("primary expr type = %s", token_to_str(node->type));
85 }
86 else if(node->expression != NULL) {
87 node->type = node->expression->type;
88 TRACE("(expression) type = %s", token_to_str(node->type));
89 }
90 else if(node->oper->type == NOT_OPER) {
91 node->type = node->right->type;
92 TRACE("not operator to type = %s", token_to_str(node->type));
93 }
94 else if(node->oper->type == UNARY_MINUS_OPER) {
95 node->type = node->right->type;
96 TRACE("negation to type = %s", token_to_str(node->type));
97 }
98 else {
99 node->type = check_expression_type(node->oper,
100 node->left->type,
101 node->right->type);
102 TRACE("%s", token_to_str(node->type));
103 }
104
105 RETURN();
106}

120static inline void set_expression_primary(ast_expr_primary_t* node) {
121
122 ENTER;
123
124 int type = node->nterm->type;
125
126 if(type == AST_RAW_VALUE) {
127 node->type = ((ast_raw_value_t*)node->nterm)->type;
128 TRACE("raw value: %s", token_to_str(node->type));
129 }
130 else if(type == AST_FUNC_REFERENCE) {
131 node->type = ((ast_func_reference_t*)node->nterm)->type;
132 TRACE("func reference: %s", token_to_str(node->type));
133 }
134 else if(type == AST_FORMATTED_STRING) {
135 node->type = STRING;
136 TRACE("formatted string: %s", token_to_str(node->type));
137 }
138 else
139 FATAL("unknown primary node type: %s", node_type_to_str(node->nterm));
140
141 RETURN();
142}

187static void pre(ast_node_t* node) {
188
189 ast_type_t type = node->type;
190
191 switch(type) {
192 case AST_FUNC_BODY_LIST:
193 ctx = ((ast_func_body_list_t*)node)->context;
194 TRACE("AST_FUNC_BODY_LIST: %p", (void*)ctx);
195 break;
196
197 case AST_LOOP_BODY_LIST:
198 ctx = ((ast_loop_body_list_t*)node)->context;
199 TRACE("AST_LOOP_BODY_LIST: %p", (void*)ctx);
200 break;
201
202 case AST_DATA_DECLARATION_LIST:
203 ctx = ((ast_data_declaration_list_t*)node)->context;
204 TRACE("AST_DATA_DECLARATION_LIST: %p", (void*)ctx);
205 break;
206
207 case AST_PROGRAM_ITEM_LIST:
208 ctx = ((ast_program_item_list_t*)node)->context;
209 TRACE("AST_PROGRAM_ITEM_LIST: %p", (void*)ctx);
210 break;
211
212 default:
213 break;
214 }

217static void post(ast_node_t* node) {
218
219 ast_type_t type = node->type;
220
221 switch(type) {
222 case AST_FUNC_DEFINITION:
223 case AST_FUNC_BODY_LIST:
224 case AST_LOOP_BODY_LIST:
225 case AST_PROGRAM_ITEM_LIST:
226 ctx = ctx->prev;
227 break;
228
229 case AST_EXPRESSION:
230 TRACE(">>> AST_EXPRESSION");
231 set_expression_type((ast_expression_t*)node);
232 TRACE("<<< AST_EXPRESSION");
233 break;
234
235 case AST_EXPR_PRIMARY:
236 TRACE("AST_EXPR_PRIMARY");
237 set_expression_primary((ast_expr_primary_t*)node);
238 break;
239
240 case AST_RAW_VALUE:
241 TRACE("AST_RAW_VALUE");
242 set_raw_value((ast_raw_value_t*)node);
243 break;
244
245 case AST_DATA_DECLARATION:
246 TRACE("AST_DATA_DECLARATION");
247 set_data_declaration((ast_data_declaration_t*)node);
248 break;
249
250 case AST_ASSIGNMENT:
251 TRACE("AST_ASSIGNMENT");
252 validate_assignment_reference((ast_assignment_t*)node);
253 break;
254
255 case AST_FUNC_REFERENCE:
256 TRACE("AST_FUNC_REFERENCE");
257 validate_func_reference((ast_func_reference_t*)node);
258 break;
259
260 case AST_ASSIGNMENT_RIGHT:
261 TRACE("AST_ASSIGNMENT_RIGHT");
262 validate_assignment_right((ast_assignment_right_t*)node);
263 break;
264
265 case AST_DATA_DEFINITION:
266 TRACE("AST_DATA_DEFINITION");
267 validate_data_definition((ast_data_definition_t*)node);
268 break;
269
270 default:
271 break;
272 }
