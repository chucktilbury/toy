27static void dump_context(context_t* ctx) {
28
29 ASSERT(ctx != NULL, "invalid context");
30
31 TRACE("this ptr: %p", (void*)ctx);
32 TRACE("prev ptr: %p", (void*)ctx->prev);
33
34 hash_table_t* tab = ctx->table;
35 TRACE("table cap: %d", tab->cap);
36 TRACE("table count: %d", tab->count);
37
38 symbol_t* sym;
39 for(int i = 0; i < tab->cap; i++) {
40 if(tab->table[i] != NULL) {
41 if(tab->table[i]->key != NULL) {
42 sym = tab->table[i]->data;
43 TRACE("symbol: %s %s %s -- %s:%s:%s:%d",
44 sym->sym_class ? "DATA" : "FUNC",
45 token_to_str(sym->sym_type), sym->name,
46 sym->is_iter ? "true" : "false",
47 sym->is_const ? "true" : "false",
48 sym->is_init ? "true" : "false",
49 sym->ref_count);
50 }
51 }
52 }
53}

95 ast_type_t type = node->type;
96
97 switch(type) {
