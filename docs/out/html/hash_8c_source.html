2/*
3 *
4 * https://programming.guide/hash-tables-open-addressing.html
5 *
6 * The hash table uses linear probing where the probing distance is
7 * hash & 0x0F. If the result is 0 then the distance is 1. When a hash
8 * is deleted, the memory is freed and the tombstone flag is set. When
9 * a hash is added, it can be added to a bucket which is a tombstone.
10 *
11 * The table is full when 3/4 of the buckets are in use. The table is
12 * resized and all of the existing hashes are rehashed into the new table.
13 *
14 * The add function tracks the max number of hops that are needed to
15 * insert a new hash. If the hops exceed a certain number, then the hash
16 * table should be rehashed, but only if a certain number of adds have
17 * taken place to avoid performance problems.
18 *
19 * Maybe tombstones should be counted instead of hops, but I do not
20 * anticipate needing to delete a lot of entries. Rehashing deletes
21 * tombstones.
22 *
23 * Test build string:
24 * clang -Wall -Wextra -g -o t hash.c memory.c errors.c
25 */

39static uint32_t hash_func(const char* key) {
40
41 uint32_t hash = 2166136261u;
42 int slen = strlen(key);
43
44 for(int i = 0; i < slen; i++) {
45 hash ^= (uint8_t)key[i];
46 hash *= 16777619;
47 }
48
49 return hash;
50}

52static int find_slot(hash_table_t* tab, const char* key) {
53
54 uint32_t hash = hash_func(key) & (tab->cap - 1);
55 int inc = hash & 0x0F;
56 inc = (inc == 0) ? 1 : inc;
57
58 if(tab->table[hash] == NULL) {
59 return hash;
60 }
61 else {
62 do {
63 for(int i = 0; i < tab->cap; i++) {
64 if(tab->table[hash] == NULL) {
65 return hash;
66 }
67 else if(tab->table[hash]->key == NULL) {
68 tab->tombstones--;
69 return hash;
70 }
71 else if(strcmp(tab->table[hash]->key, key) == 0) {
72 return hash; // duplicate key
73 }
74 else
75 hash = (hash + inc) & (tab->cap - 1);
76 }
77 inc = 1; // slot not found
78 } while(true);
79 }
80
81 return -1; // keep the compiler happy
82}

84static void rehash_table(hash_table_t* tab) {
85
86 if(tab->count * 1.75 >= tab->cap) {
87 int oldcap = tab->cap;
88 _hash_node_t** oldtab = tab->table;
89 tab->cap <<= 1; // double the capacity
90 tab->tombstones = 0;
91 tab->count = 0;
92 tab->table = _ALLOC_ARRAY(_hash_node_t*, tab->cap);
93
94 int slot;
95
96 for(int i = 0; i < oldcap; i++) {
97 if(oldtab[i] != NULL && oldtab[i]->key != NULL) {
98 slot = find_slot(tab, oldtab[i]->key);
99 tab->table[slot] = oldtab[i];
100 tab->count++;
101 }
102 }
103 _FREE(oldtab);
104 }
105}

181void remove_hashtable(hash_table_t* tab, const char* key) {
182
183 int slot = find_slot(tab, key);
184

186 if(strcmp(tab->table[slot]->key, key) == 0) {
187 _FREE(tab->table[slot]->key);
