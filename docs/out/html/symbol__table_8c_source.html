50static inline void create_data_symbol(ast_data_definition_t* node) {
51
52 SYMTAB_ENTER(AST_DATA_DEFINITION);
53
54 symbol_t* sym = create_symbol(SYM_DATA);
55 sym->is_iter = false;
56 sym->is_init = node->is_init;
57 sym->node = (ast_node_t*)node;
58
59 ast_data_declaration_t* dd = node->data_declaration;
60 sym->is_const = dd->is_const;
61 sym->sym_type = dd->type_name->token->type;
62 sym->tok = dd->IDENTIFIER;
63 sym->name = dd->IDENTIFIER->raw;
64 sym->ref_count = 0;
65
66 add_symbol(peek_context(), sym->name, sym);
67
68 TRACE("adding data symbol: %s", sym->name);
69 SYMTAB_RETURN;
70}

72static inline void create_func_params(ast_func_definition_t* node) {
73
74 SYMTAB_ENTER(AST_FUNC_DEFINITION);
75
76 ast_func_params_t* parms = node->func_params;
77 pointer_list_t* list = parms->data_declaration_list->list;
78 parms->data_declaration_list->context = create_context();
79
80 int mark = 0;
81 ast_data_declaration_t* dd;
82 while(NULL != (dd = iterate_pointer_list(list, &mark))) {
83 symbol_t* sym = create_symbol(SYM_DATA);
84 sym->is_iter = false;
85 // assignment is not legal
86 sym->is_init = true;
87 sym->is_const = true;
88 sym->node = (ast_node_t*)dd;
89
90 sym->sym_type = dd->type_name->token->type;
91 sym->tok = dd->IDENTIFIER;
92 sym->name = dd->IDENTIFIER->raw;
93 sym->ref_count = 0;
94 node->arity++;
95
96 TRACE("adding func param symbol: %s %s", token_to_str(sym->sym_type), sym->name);
97 add_symbol(peek_context(), sym->name, sym);
98
99 append_string_buffer_fmt(node->proto, "%s, ",
100 token_to_str(dd->type_name->token->type));
101 }
102
103 if(node->arity > 0)
104 trunc_string_buffer(node->proto, -2);
105 append_string_buffer_char(node->proto, ')');
106
107 SYMTAB_RETURN;
108}

110static inline void create_func_symbol(ast_func_definition_t* node) {
111
112 SYMTAB_ENTER(AST_FUNC_DEFINITION);
113
114 symbol_t* sym = create_symbol(SYM_FUNC);
115 // cannot assign to it
116 sym->is_const = true;
117 sym->is_init = true;
118 sym->node = (ast_node_t*)node;
119
120 sym->is_iter = node->func_name->is_iterator;
121 sym->sym_type = node->func_name->type_name->token->type;
122 sym->name = node->func_name->IDENTIFIER->raw;
123 sym->tok = node->func_name->IDENTIFIER;
124 sym->ref_count = 0;
125
126 add_symbol(peek_context(), sym->name, sym);
127 TRACE("adding func symbol: %s", sym->name);
128
129 node->proto = create_string_buffer(token_to_str(node->func_name->type_name->token->type));
130 append_string_buffer_fmt(node->proto, " %s(", node->func_name->IDENTIFIER->raw);
131
132 create_func_params(node);
133
134 TRACE("definition proto: %s %s:%d:%d", raw_string_buffer(node->proto),
135 sym->tok->fname, sym->tok->line_no, sym->tok->col_no);
136
137 SYMTAB_RETURN;
138}

140static void pre(ast_node_t* node) {
141
142 ast_type_t type = node->type;
143
144 switch(type) {
145 case AST_FUNC_BODY_LIST:
146 ((ast_func_body_list_t*)node)->context = create_context();
147 break;
148
149 case AST_LOOP_BODY_LIST:
150 ((ast_loop_body_list_t*)node)->context = create_context();
151 break;
152
153 case AST_PROGRAM_ITEM_LIST:
154 ((ast_program_item_list_t*)node)->context = create_context();
155 break;
156
157 case AST_DATA_DEFINITION:
158 create_data_symbol((ast_data_definition_t*)node);
159 break;
160
161 case AST_FUNC_DEFINITION:
162 create_func_symbol((ast_func_definition_t*)node);
163 break;
164
165 case AST_START_BLOCK:
166 start_block_count++;
167 break;
168
169 default:
170 break;
171 }
172}

174static void post(ast_node_t* node) {
175
176 ast_type_t type = node->type;
177
178 switch(type) {
179 case AST_FUNC_DEFINITION:
180 case AST_FUNC_BODY_LIST:
181 case AST_LOOP_BODY_LIST:
182 case AST_PROGRAM_ITEM_LIST:
183 pop_context();
184 break;
185
186 default:
187 break;
188 }

191void create_symbol_table(void) {
192
193 SEPARATOR;
194 ENTER;
195
196 if(!get_errors()) {
197 traverse_ast(pre, post);
198 }
199
200 if(start_block_count != 1) {
201 if(start_block_count == 0)
202 syntax_warning(NULL, "no start block found");
203 else if(start_block_count > 1)
204 syntax_error(NULL, "only one start block is allowed in the whole program space");
205 }
206
207 RETURN();
208}
